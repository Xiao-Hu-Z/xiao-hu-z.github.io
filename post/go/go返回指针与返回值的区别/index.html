Go语言基础之返回指针与返回值的区别

Posted by xiaohu on Thursday, October 14, 2021 
package main

import "fmt"

type A map[int]string
type B struct {
	A
	c int
}

func main() {
	b := B{make(A), 10}
	NewB := func() *B {
		return &b
	}
	c := NewB()
	c.c = 100
	c.A[1] = "3"
	fmt.Println(b, c)
}
//{map[1:3] 100} &{map[1:3] 100}

func main() {
	a := B{make(A), 10}
	b := B{make(A), 10}
	NewB := func() B {
		return b
	}
	c := NewB()
	c.c = 100
	c.A[1] = "3"
	fmt.Println(a, b, c)
}
//{map[] 10} {map[1:3] 10} {map[1:3] 100}

func main() {
	a := make(A)
	NewA := func() A {
		return a
	}
	b := NewA()
	b[1] = "10"
	fmt.Println(a, b)
}
map[1:10] map[1:10]

func main() {
	a := make(A)
	NewA := func() *A {
		return &a
	}
	b := NewA()
	b[1] = "10" //invalid operation: b[1] (type *A does not support indexing)
	fmt.Println(a, b)
}

go 语言中map,slice都是指针类型，以值的形式返回struct的时候只是把map对应的指针拷贝了
